 125    """An abstract interface for dealing with iterators.
 126
 127    This is exactly the same trait as `core::iter::Iterator` trait from Rust.
 128
 129    This is the main interface that any type can implement by basically inheriting from it.
 130    The method `__next__` is the only method that needs to be implemented, You get all the other methods for free.
 131
 132    If you want to use a ready iterator for general purposes, Use `Iter`. This interface is only for implementors
 133    and type hints.
 134
 135    Example
 136    -------
 137    ```py
 138    @dataclass
 139    class Counter(Iterator[int]):
 140        start: int = 0
 141        stop: int | None = None
 142
 143        # implement the required method.
 144        def __next__(self) -> int:
 145            result = self.start
 146            self.start += 1
 147
 148            if self.stop is not None and result >= self.stop:
 149                raise StopIteration
 150
 151            return result
 152
 153    counter = Counter(start=0, stop=10)
 154    for i in counter.map(lambda x: x * 2): # multiply each number
 155        ...
 156    ```
 157    """

 191    @typing.final
 192    def collect(
 193        self, *, cast: collections.Callable[[Item], OtherItem] | None = None
 194    ) -> collections.MutableSequence[Item] | collections.MutableSequence[OtherItem]:
 195        """Collects all items in the iterator into a sequence.
 196
 197        Example
 198        -------
 199        ```py
 200        iterator = Iter(range(3))
 201        iterator.collect()
 202        # (0, 1, 2, 3)
 203        iterator.collect(cast=str) # Map each element and collect it.
 204        # ('0', '1', '2', '3')
 205        ```
 206
 207        Parameters
 208        ----------
 209        cast: `T | None`
 210            An optional type to cast the items into.
 211            If not provided the items will be returned as it's original type.
 212        """
 213        if cast is not None:
 214            return list(cast(i) for i in self)
 215
 216        return list(_ for _ in self)

 218    @typing.final
 219    def collect_into(self, collection: Collector[Item]) -> None:
 220        """Consume this iterator, extending all items in the iterator into a mutable `collection`.
 221
 222        Example
 223        -------
 224        ```py
 225        iterator = Iter([1, 1, 2, 3, 4, 2, 6])
 226        uniques = set()
 227        iterator.collect_into(uniques)
 228        # assert uniques == {1, 2, 3, 4, 6}
 229        ```
 230
 231        Parameters
 232        ----------
 233        collection: `MutableSequence[T]` | `set[T]`
 234            The collection to extend the items in this iterator with.
 235        """
 236        if isinstance(collection, collections.MutableSequence):
 237            collection.extend(_ for _ in self)
 238        elif isinstance(collection, collections.MutableSet):
 239            collection.update(_ for _ in self)
 240        else:
 241            for idx, item in enumerate(self):
 242                collection[idx] = item

 274    @typing.final
 275    def raw_parts(self) -> collections.Generator[Item, None, None]:
 276        """Decompose this iterator into a `Generator[Item]` that yields all of the remaining items.
 277
 278        This mainly used for objects that needs to satisfy its exact type.
 279
 280        ```py
 281        it = Iter("cba")

 283
 284        assert it.count() == 0
 285        assert sort == ["a", "b", "c"]
 286        ```
 287        """
 288        for item in self:
 289            yield item

 295    def next(self) -> Option[Item]:
 296        """Advance the iterator, Returning the next item, `Some(None)` if all items yielded.
 297
 298        Example
 299        -------
 300        ```py
 301        iterator = Iter(["1", "2"])
 302        assert iterator.next() == Some("1")
 303        assert iterator.next() == Some("2")
 304        assert iterator.next().is_none()
 305        ```
 306        """
 307        try:
 308            return _option.Some(self.__next__())
 309        except StopIteration:
 310            # ! SAFETY: No more items in the iterator.
 311            return _option.NOTHING

 313    def cloned(self) -> Cloned[Item]:
 314        """Creates an iterator which shallow copies its elements by reference.
 315
 316        If you need a copy of the actual iterator and not the elements.
 317        use `Iter.clone()`
 318
 319        .. note::
 320            This method calls [`copy.copy()`](https://docs.python.org/3/library/copy.html)
 321            on each item that is being yielded.
 322
 323        Example
 324        -------
 325        ```py
 326        @dataclass
 327        class User:
 328            users_ids: list[int] = []
 329
 330        # An iterator which elements points to the same user.
 331        user = User()
 332        it = Iter((user, user))
 333
 334        for u in it.cloned():
 335            u.user_ids.append(1)
 336
 337        # We iterated over the same user pointer twice and appended "1"
 338        # since `copy` returns a shallow copy of nested structures.
 339        assert len(user.user_ids) == 2
 340        ```
 341        """
 342        return Cloned(self)
 343
 344    def copied(self) -> Copied[Item]:
 345        """Creates an iterator which copies all of its elements by value.
 346
 347        If you only need a copy of the item reference, Use `.cloned()` instead.
 348
 349        .. note::
 350            This method simply calls [`copy.deepcopy()`](https://docs.python.org/3/library/copy.html)
 351            on each item that is being yielded.
 352
 353        Example
 354        -------
 355        ```py
 356        @dataclass
 357        class User:
 358            users_ids: list[int] = []
 359
 360        # An iterator which elements points to the same user.
 361        user = User()
 362        it = Iter((user, user))
 363
 364        for u in it.copied():
 365            # A new list is created for each item.
 366            u.user_ids.append(1)
 367
 368        # The actual list is untouched since we consumed a deep copy of it.
 369        assert len(user.user_ids) == 0
 370        ```
 371        """
 372        return Copied(self)

 703    def find(self, predicate: collections.Callable[[Item], bool]) -> Option[Item]:
 704        """Searches for an element of an iterator that satisfies a predicate.
 705
 706        If you want the position of the element, use `Iterator.position` instead.
 707
 708        `find()` takes a lambda that returns true or false. It applies this closure to each element of the iterator,
 709        and if any of them return true, then find() returns `Some(element)`. If they all return false, it returns None.
 710
 711        Example
 712        -------
 713        ```py
 714        it = Iter(range(10))
 715        item = it.find(lambda num: num > 5)
 716        print(item) # 6
 717        ```
 718        """
 719        for item in self:
 720            if predicate(item):
 721                return _option.Some(item)
 722
 723        # no more items
 724        return _option.NOTHING

 726    def position(self, predicate: collections.Callable[[Item], bool]) -> Option[int]:
 727        """Searches for the position of an element in the iterator that satisfies a predicate.
 728
 729        If you want the object itself, use `Iterator.find` instead.
 730
 731        `position()` takes a lambda that returns true or false. It applies this closure to each element of the iterator,
 732        and if any of them return true, then position() returns `Some(position_of_element)`. If they all return false, it returns None.
 733
 734        Example
 735        -------
 736        ```py
 737        it = Iter(range(10))
 738        position = it.find(lambda num: num > 5)
 739        assert position.unwrap() == 6
 740        ```
 741        """
 742        for position, value in self.enumerate():
 743            if predicate(value):
 744                return _option.Some(position)
 745
 746        # no more items
 747        return _option.NOTHING

 749    def fold(
 750        self, init: OtherItem, f: collections.Callable[[OtherItem, Item], OtherItem]
 751    ) -> OtherItem:
 752        """Folds every element into an accumulator by applying an operation, returning the final result.
 753
 754        fold() takes two arguments: an initial value, and a closure with two arguments: an accumulator, and an element.
 755        The closure returns the value that the accumulator should have for the next iteration.
 756
 757        The initial value is the value the accumulator will have on the first call.
 758
 759        After applying this closure to every element of the iterator, fold() returns the accumulator.
 760
 761        This operation is sometimes called reduce or inject.
 762
 763        Example
 764        -------
 765        ```py
 766        a = Iter([1, 2, 3, 4])
 767        sum = a.fold(0, lambda acc, elem: acc + elem)
 768        assert sum == 10
 769        ```
 770        """

An abstract interface for dealing with iterators.

This is exactly the same trait as core::iter::Iterator trait from Rust.

This is the main interface that any type can implement by basically inheriting from it. The method __next__ is the only method that needs to be implemented, You get all the other methods for free.

If you want to use a ready iterator for general purposes, Use Iter. This interface is only for implementors and type hints.

Example
@dataclass
class Counter(Iterator[int]):
    start: int = 0
    stop: int | None = None

    # implement the required method.
    def __next__(self) -> int:
        result = self.start
        self.start += 1

        if self.stop is not None and result >= self.stop:
            raise StopIteration

        return result

counter = Counter(start=0, stop=10)
for i in counter.map(lambda x: x * 2): # multiply each number
    ...

Returns an iterator that yields the exact same element number of count times.

The yielded elements is a copy of element, but the last element is guaranteed to be the same as the original element.

Example
nums = [1, 2, 3]
it = sain.iter.repeat(nums, 5)
for i in range(4):
    cloned = it.next().unwrap()
    assert cloned == [1, 2, 3]

# But the last item is the origin one...
last = it.next().unwrap()
last.append(4)
assert nums == [1, 2, 3, 4]
